    public procedure addMembre(m : Membre)
    Debut
        si (this.nbrMembre >= N) alors
            ecrire("Nombre maximum de membres atteint.")            
        Sinon
            this.nbrMembre <-- this.nbrMembre + 1
            m.setGroupe(this)
            this.membres[this.nbrMembre] <-- m
        finsi
    Fin

    public static fonction estMembre(u : User) : boolean 
    var i : entier
    Debut
        pour (i <-- 1; i <= this.nbrMembre; i++) faire
            si (this.membres[i].getUser().getNomUtilisateur() = u.getNomUtilisateur()) alors
                retourne Vrai
            finsi
        finpour
        retourne Faux
    Fin
